== for

[source,bash]
.配列を処理
----
#!/bin/bash
for var in hoge fuga foo bar
do
    echo $var
done
----

[source,bash]
.10から20まで+2ずつ処理
----
#!/bin/bash
for var in `seq 10 2 20`
do
    echo $var
done
# 10, 12, 14, 16, 18, 20
----

[source,bash]
.loop でぐるぐるスクリプト
----
#!/bin/bash
high=(aaa bbb ccc)
middle=(iii jjj kkk)
low=(xxx yyy zzz)
for i in ${high[@]};
do
  for j in ${middle[@]};
  do
    for k in ${low[@]};
    do
      name=${i}_${j}_${k}
      echo ${name}
      mkdir ${name}
      json=$(cat << EOS
{
  "high": "${i}",
  "middle": "${j}",
  "low": "${k}"
}
EOS
)
      echo ${json} > ${name}/${name}.json
      zip -r ${name}.zip ${name}
      rm -rf ${name}
    done
  done
done
----
